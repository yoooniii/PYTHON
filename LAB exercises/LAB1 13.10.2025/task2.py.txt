#Чете от конзолата число R(радиус) и намира лиицето и периметъра на окръжност, като резултатът закръгляме до 3ти знак.
#S=pi r^2
#P=2pi r

r = float(input("Въведете радиус r: "))
pi = 3.14159
S = pi * r * r
P = 2 * pi * r
print(f"Лицето на окръжността е: {round(S, 3)}")
print(f"Периметърът е: {round(P, 3)}")
